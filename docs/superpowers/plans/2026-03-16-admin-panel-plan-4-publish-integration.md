# Admin Panel Plan 4: Publish Pipeline + Public Site Integration

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Connect the admin panel to the public site by implementing the JSON export, publish workflow (build + deploy), and updating the Astro components to read from exported data instead of hardcoded files.

**Architecture:** The admin server exports the database to `src/data/content.json`. The publish endpoint runs Astro build and FTP deploy. Astro components read from `content.json` at build time. Apply forms POST to the admin API.

**Tech Stack:** Node.js child_process for build/deploy, Astro data loading, existing deploy.sh

**Spec:** `docs/superpowers/specs/2026-03-16-admin-panel-design.md`

**Depends on:** Plans 1-3

```typescript
import db from './db.js';
import fs from 'node:fs';
import path from 'node:path';
import { fileURLToPath } from 'node:url';

const __dirname = path.dirname(fileURLToPath(import.meta.url));
const EXPORTS_DIR = path.join(__dirname, '..', 'data', 'exports');
const ASTRO_DATA_DIR = path.join(__dirname, '..', '..', 'src', 'data');

export function exportData(): string {
  fs.mkdirSync(EXPORTS_DIR, { recursive: true });
  fs.mkdirSync(ASTRO_DATA_DIR, { recursive: true });

  // Get published projects with photos
  const projects = db.prepare(
    "SELECT * FROM projects WHERE status = 'published' ORDER BY displayOrder ASC"
  ).all() as any[];

  const projectsWithPhotos = projects.map(p => {
    const photos = db.prepare(
      'SELECT * FROM project_photos WHERE projectId = ? ORDER BY displayOrder ASC'
    ).all(p.id) as any[];
    return {
      ...p,
      services: JSON.parse(p.services || '[]'),
      photos: photos.map(ph => ({
        ...ph,
        serviceTags: JSON.parse(ph.serviceTags || '[]'),
      })),
    };
  });

  // Get active job listings
  const jobs = db.prepare(
    'SELECT * FROM job_listings WHERE isActive = 1 ORDER BY department ASC, displayOrder ASC'
  ).all() as any[];

  const content = {
    projects: projectsWithPhotos,
    jobs,
    exportedAt: new Date().toISOString(),
  };

  const json = JSON.stringify(content, null, 2);

  // Write to exports dir (backup)
  fs.writeFileSync(path.join(EXPORTS_DIR, 'content.json'), json);

  // Write to Astro src/data for build
  fs.writeFileSync(path.join(ASTRO_DATA_DIR, 'content.json'), json);

  return content.exportedAt;
}
```

```typescript
import { Router, Request, Response } from 'express';
import { exec } from 'node:child_process';
import path from 'node:path';
import { fileURLToPath } from 'node:url';
import { requireAuth } from '../auth.js';
import { exportData } from '../export.js';

const __dirname = path.dirname(fileURLToPath(import.meta.url));
const PROJECT_ROOT = path.join(__dirname, '..', '..', '..');

const router = Router();

let publishState = {
  status: 'idle' as 'idle' | 'exporting' | 'building' | 'deploying' | 'live' | 'error',
  lastPublish: null as string | null,
  error: null as string | null,
};

// GET /api/publish/status
router.get('/status', requireAuth, (_req: Request, res: Response) => {
  res.json(publishState);
});

// POST /api/publish
router.post('/', requireAuth, async (_req: Request, res: Response) => {
  if (publishState.status !== 'idle' && publishState.status !== 'live' && publishState.status !== 'error') {
    res.status(409).json({ error: 'Publish already in progress' });
    return;
  }

  // Start publish process
  res.json({ message: 'Publish started' });

  try {
    // Step 1: Export data
    publishState = { ...publishState, status: 'exporting', error: null };
    exportData();

    // Step 2: Build
    publishState = { ...publishState, status: 'building' };
    await runCommand('npm run build', PROJECT_ROOT);

    // Step 3: Deploy
    publishState = { ...publishState, status: 'deploying' };
    await runCommand('bash deploy.sh', PROJECT_ROOT);

    // Done
    const now = new Date().toISOString();
    publishState = { status: 'live', lastPublish: now, error: null };

    // Reset to idle after 10 seconds
    setTimeout(() => {
      if (publishState.status === 'live') {
        publishState = { ...publishState, status: 'idle' };
      }
    }, 10000);
  } catch (err: any) {
    publishState = { ...publishState, status: 'error', error: err.message };
  }
});

function runCommand(cmd: string, cwd: string): Promise<string> {
  return new Promise((resolve, reject) => {
    exec(cmd, { cwd, timeout: 300000 }, (error, stdout, stderr) => {
      if (error) {
        reject(new Error(`${cmd} failed: ${stderr || error.message}`));
        return;
      }
      resolve(stdout);
    });
  });
}

export default router;
```

- [ ] **Step 2: Add form submission to apply-employee.astro**

Add a script at the bottom of the page that intercepts form submission and POSTs to the admin API:

```html
<script>
  const form = document.querySelector('form');
  if (form) {
    form.addEventListener('submit', async (e) => {
      e.preventDefault();
      const formData = new FormData(form);

      // Collect all form fields into a JSON object
      const data = {};
      formData.forEach((value, key) => {
        if (key !== 'resume') data[key] = value;
      });

      const submitData = new FormData();
      submitData.append('name', formData.get('name') || formData.get('firstName') + ' ' + formData.get('lastName'));
      submitData.append('email', formData.get('email'));
      submitData.append('phone', formData.get('phone') || '');
      submitData.append('jobListingId', formData.get('position') || '');
      submitData.append('formData', JSON.stringify(data));

      const resume = formData.get('resume');
      if (resume instanceof File && resume.size > 0) {
        submitData.append('resume', resume);
      }

      try {
        const res = await fetch('https://admin.qualityreflections.com/api/applications', {
          method: 'POST',
          body: submitData,
        });
        if (res.ok) {
          // Show success message
          form.innerHTML = '<div class="text-center py-12"><h3 class="text-xl mb-2" style="color: var(--glass-blue);">Application Submitted!</h3><p style="color: var(--silver);">Thank you for your interest. We will review your application and get back to you.</p></div>';
        } else {
          throw new Error('Submission failed');
        }
      } catch {
        alert('There was an error submitting your application. Please try again or contact us directly.');
      }
    });
  }
</script>
```
